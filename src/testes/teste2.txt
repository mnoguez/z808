EMPILHA MACRO REG1,REG2,REG3,REG4,REG5,REG6,REG7
IFNB <REG1>
MOV AX,REG1
PUSH AX
EMPILHA REG2,REG3,REG4,REG5,REG6,REG7
ENDIF
ENDM

DIVIDE MACRO NUMERADOR,DENOMINADOR
IFDIF <DENOMINADOR>,<0>
MOV AX,0
MOV DX,AX
MOV AX,DENOMINADOR
MOV SI,AX
MOV AX,NUMERADOR
ENDIF
ENDM

ALOCAR MACRO NROPALAVRAS
CONTADOR=0 
REPT NROPALAVRAS
IFDIF <CONTADOR>,<NROPALAVRAS>
DW ?
ELSE
EXITM
ENDIF
CONTADOR=CONTADOR+1
ENDM

DADOS SEGMENT
VETOR DW ?
ALOQUEMEM 4
VAR1 DW 8
VAR2 DW 3
DADOS ENDS

CODIGO SEGMENT
ASSUME CS:CODIGO
ASSUME DS:DADOS
DIVIDE VAR1,VAR2
DIVIDE VAR1,0
EMPILHA AX,DX,SI,CS,DS,SS,SP
EMPILHA VAR1,VAR2
CODIGO ENDS
END
